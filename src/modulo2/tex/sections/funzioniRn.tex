\section{Funzioni in $\mathbb{R}^n$}
\dfn{
	Dato un \textbf{centro} $x \in \mathbb{R}^n$ e un \textbf{raggio} $r > 0$ poniamo 
	\begin{equation*}
		B(x, r) = \{y \in \mathbb{R}^n \; | \; \lVert y - x \rVert < r\}
	\end{equation*}
	$B$ si chiama \textbf{disco di centro $\mathbf{x}$ e raggio $\mathbf{r}$}
}
